    function blendWithNeutral(hex, intensity) {
        const rv = parseInt(hex.slice(1,3), 16);
        const gv = parseInt(hex.slice(3,5), 16);
        const bv = parseInt(hex.slice(5,7), 16);
        const t  = intensity / 100;
        const n  = 12;
        const ri = Math.round(n + (rv - n) * t);
        const gi = Math.round(n + (gv - n) * t);
        const bi = Math.round(n + (bv - n) * t);
        return `#${ri.toString(16).padStart(2,'0')}${gi.toString(16).padStart(2,'0')}${bi.toString(16).padStart(2,'0')}`;
    }

    function drawWheel() {
        const cx = 25, cy = 25, r = 20;
        wctx.clearRect(0, 0, 50, 50);
        wctx.beginPath(); wctx.arc(cx, cy, r, 0, Math.PI * 2);
        wctx.fillStyle = 'rgba(255,255,255,.06)'; wctx.fill();
        wctx.strokeStyle = 'rgba(255,255,255,.18)'; wctx.lineWidth = 1; wctx.stroke();
        const rad = currentGradAngle * Math.PI / 180;
        const ex = cx + r * Math.sin(rad), ey = cy - r * Math.cos(rad);
        wctx.beginPath(); wctx.moveTo(cx, cy); wctx.lineTo(ex, ey);
        wctx.strokeStyle = 'rgba(255,255,255,.75)'; wctx.lineWidth = 1.5; wctx.stroke();
        wctx.beginPath(); wctx.arc(ex, ey, 3.5, 0, Math.PI * 2);
        wctx.fillStyle = '#fff'; wctx.fill();
        wctx.beginPath(); wctx.arc(cx, cy, 2, 0, Math.PI * 2);
        wctx.fillStyle = 'rgba(255,255,255,.3)'; wctx.fill();
    }
    drawWheel();

    function angleFromPointer(e) {
        const rect = wheelCanvas.getBoundingClientRect();
        const dx = e.clientX - (rect.left + 25), dy = e.clientY - (rect.top + 25);
        let a = Math.round(Math.atan2(dx, -dy) * 180 / Math.PI);
        if (a < 0) a += 360;
        return a % 360;
    }

        function initModdingTools() {
            const main = document.querySelector('body > div.wrapper > div.centerpanel');
            if (!main) return;
            const terminal = main.querySelector('.runpanelcontainer');
            const editor   = main.querySelector('.editorpanel');
            if (!terminal || !editor || !editor.childNodes[1]) return;
            const bar   = editor.childNodes[1];
            const consl = terminal.childNodes[1] && terminal.childNodes[1].childNodes[1];
            if (!consl) return;
            if (document.getElementById('runstopmod')) return; // already initialized

            // --- Run / Terminate Mod button (replaces original #runmod) ---
            const origRun = document.getElementById('runmod');
            if (origRun) origRun.remove();

            const runBtn = document.createElement('a');
            runBtn.id = 'runstopmod';
            runBtn.href = '#';
            runBtn.setAttribute('data-tooltip', 'Run Mod');
            runBtn.setAttribute('cmd', 'start');
            runBtn.innerHTML = '<i class="fa fa-fw fa-play"></i>';
            runBtn.addEventListener('click', e => {
                e.preventDefault();
                try { window.$('#terminal').terminal().exec(runBtn.getAttribute('cmd'), true); } catch(_) {}
            });
            bar.appendChild(runBtn);

            // --- Open Game Frame button (hidden until mod is running) ---
            const testBtn = document.createElement('a');
            testBtn.id = 'test';
            testBtn.href = '#';
            testBtn.style.display = 'none';
            testBtn.setAttribute('data-tooltip', 'Open Game Frame');
            testBtn.innerHTML = '<i class="fa fa-fw fa-gamepad"></i>';
            testBtn.addEventListener('click', e => {
                e.preventDefault();
                try { window.$('#terminal').terminal().exec('test', true); } catch(_) {}
            });
            bar.appendChild(testBtn);

            // --- Region selector ---
            const regions = ['Asia', 'America', 'Europe'];
            const regc = document.createElement('select');
            regc.id = 'region-select';
            regc.innerHTML = '<option disabled>Select region</option>' +
                regions.map(r => `<option value="${r}">${r}</option>`).join('');
            try {
                const termOut = document.querySelector('.terminal-output');
                if (termOut) {
                    const div3 = termOut.getElementsByTagName('div')[3];
                    if (div3) {
                        const regionText = div3.innerText.replace(/Region\sset\sto\s/g, '').trim();
                        const idx = regions.indexOf(regionText);
                        if (idx !== -1) regc.options.selectedIndex = idx + 1;
                    }
                }
            } catch(_) {}
            regc.addEventListener('change', () => {
                try { window.$('#terminal').terminal().exec('region ' + regc.value, true); } catch(_) {}
            });
            consl.appendChild(regc);

            // --- Clear Console button ---
            const clearBtn = document.createElement('a');
            clearBtn.href = '#';
            clearBtn.setAttribute('data-tooltip', 'Clear Console');
            clearBtn.innerHTML = '<i class="fa fa-fw fa-trash-o"></i>';
            clearBtn.addEventListener('click', e => {
                e.preventDefault();
                try { window.$('#terminal').terminal().clear(); } catch(_) {}
            });
            consl.appendChild(clearBtn);

            // --- Hide Console Panel / Hide Editor Panel buttons ---
            const hconsl = document.createElement('a');
            hconsl.id = 'Console';
            hconsl.href = '#';
            hconsl.style.float = 'left';
            hconsl.setAttribute('data-tooltip', 'Hide Console Panel');
            hconsl.innerHTML = '<i class="fa fa-fw fa-forward"></i>';

            const heditorBtn = document.createElement('a');
            heditorBtn.id = 'Editor';
            heditorBtn.href = '#';
            heditorBtn.style.float = 'right';
            heditorBtn.setAttribute('data-tooltip', 'Hide Editor Panel');
            heditorBtn.innerHTML = '<i class="fa fa-fw fa-backward"></i>';

            const setoption = element => {
                const htext  = ['Show', 'Hide'];
                const hpos   = ['Console', 'Editor'];
                const parts  = element.getAttribute('data-tooltip').split(' ');
                const id     = element.getAttribute('id');
                const d      = Math.abs(hpos.indexOf(id || ''));
                let elempos  = [terminal, editor];
                const isHide = Math.abs(htext.indexOf(parts[0] || ''));
                let elem     = heditorBtn;

                if (isHide) {
                    if (d) {
                        elempos = [editor, terminal];
                        elem = hconsl;
                        bar.style.display = 'none';
                    } else {
                        const fv = document.getElementById('fieldview');
                        if (fv) fv.style.display = 'none';
                    }
                    elempos[0].style.width = '0%';
                    elempos[1].style.width = '100%';
                } else {
                    elempos.forEach(el => el.removeAttribute('style'));
                    const fv = document.getElementById('fieldview');
                    if (fv) fv.removeAttribute('style');
                    bar.removeAttribute('style');
                    elem = element;
                }

                try { ace.edit('editor').resize(); } catch(_) {}
                elem.setAttribute('data-tooltip',
                    [htext[Number(!isHide)], hpos[Number(d)], 'Panel'].join(' '));
            };

            hconsl.addEventListener('click',     e => { e.preventDefault(); setoption(hconsl); });
            heditorBtn.addEventListener('click', e => { e.preventDefault(); setoption(heditorBtn); });

            bar.appendChild(heditorBtn);
            consl.insertBefore(hconsl, consl.childNodes[0]);

            // --- Hook Modding prototype (state sync + beforeunload guard) ---
            const hookModding = () => {
                if (typeof window.Modding === 'undefined') return setTimeout(hookModding, 250);

                window.onbeforeunload = () => {
                    if (window.game && window.game.modding && window.game.modding.field_view)
                        return 'Mod is still running. Leave anyway?';
                    return null;
                };

                const alternate = name => {
                    const ct     = ['stop', 'start'];
                    const cmd    = ct.indexOf(name);
                    const titles = ['Run', 'Terminate'];
                    const icons  = ['play', 'stop'];
                    const run    = document.getElementById('runstopmod');
                    if (!run || cmd === -1) return;
                    run.setAttribute('cmd', ct[Math.abs(cmd - 1)]);
                    run.setAttribute('data-tooltip', titles[cmd] + ' Mod');
                    const icon = run.querySelector('i');
                    if (icon) icon.className = 'fa fa-fw fa-' + icons[cmd];
                };

                try {
                    const _stopped    = Modding.prototype.stopped;
                    const _modStarted = Modding.prototype.modStarted;
                    const _setRegion  = Modding.prototype.setRegion;

                    Modding.prototype.stopped = function(...args) {
                        alternate('stop');
                        const t = document.getElementById('test');
                        if (t) t.style.display = 'none';
                        return _stopped.call(this, ...args);
                    };
                    Modding.prototype.modStarted = function(...args) {
                        alternate('start');
                        const t = document.getElementById('test');
                        if (t) t.removeAttribute('style');
                        return _modStarted.call(this, ...args);
                    };
                    Modding.prototype.setRegion = function(...args) {
                        const result = _setRegion.call(this, ...args);
                        const sel = document.getElementById('region-select');
                        if (sel && args[0]) {
                            const idx = regions.indexOf(String(args[0]));
                            if (idx !== -1) sel.options.selectedIndex = idx + 1;
                        }
                        return result;
                    };
                } catch(err) {
                    console.warn('[ThemeScript] Modding prototype hook failed:', err);
                }
            };
            hookModding();
        }

        // Wait for the editor DOM to be ready before inserting tools
        const editorReady = () =>
            document.querySelector('body > div.wrapper > div.centerpanel > .editorpanel');
        if (editorReady() && editorReady().childNodes[1]) {
            initModdingTools();
        } else {
            const obs = new MutationObserver(() => {
                if (editorReady() && editorReady().childNodes[1]) {
                    obs.disconnect();
                    initModdingTools();
                }
            });
            obs.observe(document.body || document.documentElement, { childList: true, subtree: true });
        }
    }

})();
